    <script>
        document.addEventListener("DOMContentLoaded", function() {
            const darkModeToggle = document.getElementById("toggleDarkMode");
            const refreshButton = document.getElementById("refreshData");
            const rsiFilter = document.getElementById("rsiFilter");
            const stockContainer = document.getElementById("stockContainer");

            if (localStorage.getItem("darkMode") === "enabled") {
                document.body.classList.add("dark-mode");
                darkModeToggle.textContent = "â˜€ï¸ Light Mode";
            }

            darkModeToggle.addEventListener("click", function() {
                document.body.classList.toggle("dark-mode");
                if (document.body.classList.contains("dark-mode")) {
                    localStorage.setItem("darkMode", "enabled");
                    darkModeToggle.textContent = "â˜€ï¸ Light Mode";
                } else {
                    localStorage.setItem("darkMode", "disabled");
                    darkModeToggle.textContent = "ðŸŒ™ Dark Mode";
                }
            });

            refreshButton.addEventListener("click", function() {
                location.reload();
            });

            rsiFilter.addEventListener("change", function() {
                const filterValue = rsiFilter.value;
                document.querySelectorAll(".stock-item").forEach(stock => {
                    const signal = stock.getAttribute("data-signal");
                    if (filterValue === "all" || signal.includes(filterValue)) {
                        stock.style.display = "block";
                    } else {
                        stock.style.display = "none";
                    }
                });
            });
        });
    </script>

    <script>
const historicalData = {{ historical_data | tojson }};

Object.entries(historicalData).forEach(([symbol, data]) => {
    const canvas = document.getElementById(`chart_${symbol}`);
    if (!canvas) return;

    new Chart(canvas, {
        type: 'line',
        data: {
            labels: data.dates,
            datasets: [
                {
                    label: 'Close Price',
                    data: data.close_prices,
                    borderColor: 'rgba(75, 192, 192, 1)',
                    backgroundColor: 'rgba(75, 192, 192, 0.2)',
                    borderWidth: 2,
                    fill: false,
                    tension: 0.3
                },
                {
                    label: 'RSI',
                    data: data.rsi,
                    borderColor: 'rgba(255, 99, 132, 1)',
                    borderWidth: 1,
                    borderDash: [5, 5],
                    fill: false,
                    tension: 0.2
                }
            ]
        },
        options: {
            responsive: true,
            plugins: {
                legend: {
                    position: 'top'
                },
                title: {
                    display: true,
                    text: symbol
                }
            },
            scales: {
                y: {
                    beginAtZero: false
                }
            }
        }
    });

    // Export buttons
    const btns = document.createElement('div');
    btns.innerHTML = `
        <button class="btn btn-sm btn-outline-primary mt-2 export-btn" data-type="png" data-symbol="${symbol}">ðŸ–¼ Export PNG</button>
        <button class="btn btn-sm btn-outline-danger mt-2 ml-2 export-btn" data-type="pdf" data-symbol="${symbol}">ðŸ“„ Export PDF</button>
    `;
    canvas.parentElement.appendChild(btns);
});

// Export Logic
document.addEventListener('click', function(e) {
    if (e.target.classList.contains('export-btn')) {
        const type = e.target.getAttribute('data-type');
        const symbol = e.target.getAttribute('data-symbol');
        const canvas = document.getElementById(`chart_${symbol}`);
        if (type === 'png') {
            const img = canvas.toDataURL("image/png");
            const link = document.createElement("a");
            link.href = img;
            link.download = `${symbol}_chart.png`;
            link.click();
        } else if (type === 'pdf') {
            const imgData = canvas.toDataURL('image/jpeg', 1.0);
            const pdf = new jspdf.jsPDF();
            pdf.addImage(imgData, 'JPEG', 15, 15, 180, 100);
            pdf.save(`${symbol}_chart.pdf`);
        }
    }
});
